017package grails.plugins.quartz;
018
019import java.beans.PropertyEditorSupport;
020import java.text.ParseException;
021import java.util.Date;
022import java.util.Map;
023
024import org.quartz.JobDetail;
025import org.quartz.Trigger;
026import org.quartz.impl.triggers.AbstractTrigger;
027import org.springframework.beans.BeanUtils;
028import org.springframework.beans.BeanWrapper;
029import org.springframework.beans.PropertyAccessorFactory;
030import org.springframework.beans.factory.FactoryBean;
031import org.springframework.beans.factory.InitializingBean;
032
033/**
034 * The factory bean to create and register trigger beans in Spring context.
035 *
036 * @author Sergey Nebolsin (nebolsin@gmail.com)
037 */
038public class CustomTriggerFactoryBean implements FactoryBean<Trigger>, InitializingBean {
039    private Class<Trigger> triggerClass;
040    private Trigger customTrigger;
041    private JobDetail jobDetail;
042
043    private Map triggerAttributes;
044
045    public void afterPropertiesSet() throws ParseException {
046        // Create a trigger by the class name
047        customTrigger = BeanUtils.instantiateClass(triggerClass);
048
049        // If trigger is a standard trigger, set standard properties
050        if(customTrigger instanceof AbstractTrigger){
051            AbstractTrigger at =(AbstractTrigger) customTrigger;
052
053            // Set job details
054            if(jobDetail!=null){
055                at.setJobKey(jobDetail.getKey());
056            }
057
058            // Set start delay
059            if (triggerAttributes.containsKey(GrailsJobClassConstants.START_DELAY)) {
060                Number startDelay = (Number) triggerAttributes.remove(GrailsJobClassConstants.START_DELAY);
061                at.setStartTime(new Date(System.currentTimeMillis() + startDelay.longValue()));
062            }
063        }
064
065        // Set non standard properties.
066        BeanWrapper customTriggerWrapper = PropertyAccessorFactory.forBeanPropertyAccess(customTrigger);
067        customTriggerWrapper.registerCustomEditor(String.class, new StringEditor());
068        customTriggerWrapper.setPropertyValues(triggerAttributes);
069    }
070
071    /**
072     * {@inheritDoc}
073     *
074     * @see org.springframework.beans.factory.FactoryBean#getObject()
075     */
076    public Trigger getObject() throws Exception {
077        return customTrigger;
078    }
079
080    /**
081     * {@inheritDoc}
082     *
083     * @see org.springframework.beans.factory.FactoryBean#getObjectType()
084     */
085    public Class<? extends Trigger> getObjectType() {
086        return triggerClass;
087    }
088
089    /**
090     * {@inheritDoc}
091     *
092     * @see org.springframework.beans.factory.FactoryBean#isSingleton()
093     */
094    public boolean isSingleton() {
095        return true;
096    }
097
098    public void setJobDetail(JobDetail jobDetail) {
099        this.jobDetail = jobDetail;
100    }
101
102    public void setTriggerClass(Class<Trigger> triggerClass) {
103        this.triggerClass = triggerClass;
104    }
105
106    public void setTriggerAttributes(Map triggerAttributes) {
107        this.triggerAttributes = triggerAttributes;
108    }
109}
110
111// We need this additional editor to support GString -> String convertion for trigger's properties.
112class StringEditor extends PropertyEditorSupport {
113    @Override
114    public void setValue(Object value) {
115        super.setValue(value == null ? null : value.toString());
116    }
117
118    @Override
119    public void setAsText(String text) throws IllegalArgumentException {
120        setValue(text);
121    }
122}
